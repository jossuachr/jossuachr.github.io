# Load Libraries
library(tidyverse)
library(randomForest)
library(scales)
library(lubridate)
library(ranger)
library(iml)
library(sf)
library(leaflet)
library(shiny)

# Read Data
Unemployment <- read.csv("~/Downloads/Unemployment.csv", stringsAsFactors = TRUE)
PriceAvailability <- read.csv("~/Downloads/Price and Availability Data.csv", stringsAsFactors = TRUE)
MajorMarket <- read.csv("~/Downloads/Major Market Occupancy Data-revised.csv", stringsAsFactors = TRUE)
Leases <- read.csv("~/Downloads/Leases.csv", stringsAsFactors = TRUE)

# Merge Datasets
Leases <- merge(Leases, Unemployment, by = c("year", "quarter", "state"), all.x = TRUE)
Leases <- merge(Leases, MajorMarket, by = c("year", "quarter", "market"), all.x = TRUE)

# Mode Function
get_mode <- function(x) {
  ux <- unique(x);
  ux[which.max(tabulate(match(x, ux)))]
}

# Group Summarize
grouped_data <- Leases %>%
  group_by(overall_rent) %>%
  summarize(
    unemployment_rate = mean(unemployment_rate, na.rm = TRUE),
    avg_occupancy_proportion = mean(avg_occupancy_proportion, na.rm = TRUE),
    leasedSF = mean(leasedSF, na.rm = TRUE),
    RBA = mean(RBA, na.rm = TRUE),
    availability_proportion = mean(availability_proportion, na.rm = TRUE),
    market = get_mode(market),
    zip = get_mode(zip),
    internal_class = get_mode(internal_class),
    transaction_type = get_mode(transaction_type),
    space_type = get_mode(space_type),
    CBD_suburban = get_mode(CBD_suburban),
    year = get_mode(year),
    monthsigned = get_mode(monthsigned),
    .groups = 'drop'
  )
head(grouped_data)

# EDA Plots
# Example: Histogram of overall rent
ggplot(grouped_data, aes(x = overall_rent)) +
  geom_histogram(bins = 30, fill = 'steelblue', alpha = 0.8) +
  labs(title = 'Distribution of Overall Rent', x = 'Overall Rent', y = 'Count')

# Random Forest Modeling
set.seed(123)
split_data <- initial_split(grouped_data, strata = overall_rent)
train_data <- training(split_data)
test_data <- testing(split_data)

rf_spec <- rand_forest(
  mode = 'regression',
  trees = 1000,
  mtry = tune(),
  min_n = tune()
) %>% set_engine('ranger')

rf_recipe <- recipe(overall_rent ~ unemployment_rate + avg_occupancy_proportion + leasedSF + RBA + availability_proportion,
                    data = train_data) %>%
  step_impute_median(all_numeric(), -all_outcomes()) %>%
  step_normalize(all_numeric(), -all_outcomes())

rf_wf <- workflow() %>%
  add_model(rf_spec) %>%
  add_recipe(rf_recipe)

rf_grid <- grid_random(mtry(range = c(2, 5)), min_n(range = c(5, 20)), size = 20)
cv <- vfold_cv(train_data, v = 5)

tune_res <- tune_grid(rf_wf, resamples = cv, grid = rf_grid, metrics = metric_set(rmse, rsq))

best_params <- select_best(tune_res, 'rmse')
final_wf <- finalize_workflow(rf_wf, best_params)
final_fit <- last_fit(final_wf, split_data)

# Variable Importance & ALE
top_vip <- vip::vip(extract_fit_parsnip(final_wf %>% fit(train_data)))
ale_income <- FeatureEffect$new(extract_fit_parsnip(final_wf %>% fit(train_data)), 'unemployment_rate', method = 'ale')

# Geospatial Mapping
us_map <- MajorMarket %>%
  filter(state %in% c('GA', 'TX', 'MD', 'MA')) %>%
  group_by(market) %>%
  summarize(avg_rent = mean(overall_rent, na.rm = TRUE)) %>%
  left_join(states_sf, by = 'state') %>%
  st_as_sf()
leaflet(us_map) %>% addTiles() %>% addPolygons(fillColor = ~pal(avg_rent))
