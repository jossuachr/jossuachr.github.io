# Load libraries
library(tidyverse)
library(randomForest)
library(scales)
library(lubridate)
library(ranger)
library(iml)
library(sf)
library(leaflet)
library(shiny)

# Read data
df_unemp  <- read.csv("assets/Unemployment.csv", stringsAsFactors=TRUE)
df_price  <- read.csv("assets/PriceAndAvailability.csv", stringsAsFactors=TRUE)
df_market <- read.csv("assets/MajorMarket.csv", stringsAsFactors=TRUE)
df_leases <- read.csv("assets/Leases.csv", stringsAsFactors=TRUE)

# Merge datasets
leases_full <- df_leases %>%
  merge(df_unemp, by = c("year","quarter","state"), all.x=TRUE) %>%
  merge(df_market, by = c("year","quarter","market"), all.x=TRUE)

# Clean & feature engineer
leases_clean <- leases_full %>%
  group_by(lease_id) %>%
  summarize(overall_rent = sum(rent_amount), .groups='drop') %>%
  left_join(leases_full, by="lease_id") %>%
  mutate(price_category = ifelse(predicted_rent > overall_rent, "Under", "Over"))

# Train-test split
set.seed(123)
split_data <- initial_split(leases_clean, strata=price_category)
train_data <- training(split_data)
test_data  <- testing(split_data)

# Model spec & tuning
rf_spec <- rand_forest(mode="regression", trees=1000, mtry=tune(), min_n=tune()) %>%
  set_engine("ranger")

market_recipe <- recipe(overall_rent ~ median_income + unemployment_rate + lease_duration_months + market_size,
                         data=train_data) %>%
  step_impute_median(all_numeric()) %>%
  step_dummy(all_nominal()) %>%
  step_normalize(all_numeric())

wf <- workflow() %>%
  add_recipe(market_recipe) %>%
  add_model(rf_spec)

rf_grid <- grid_regular(mtry(range=c(2,5)), min_n(range=c(5,20)), levels=4)

set.seed(456)
rf_tune <- tune_grid(wf, resamples=vfold_cv(train_data, v=5), grid=rf_grid, metrics=metric_set(rmse, rsq))

best_params <- rf_tune %>% select_best("rmse")
final_wf <- finalize_workflow(wf, best_params)
final_fit <- final_wf %>% last_fit(split_data)

# Interpretability
vip(final_fit$.workflow[[1]] %>% extract_fit_parsnip())
ale_income <- FeatureEffect$new(final_fit$.workflow[[1]] %>% extract_fit_parsnip(), feature="median_income", method="ale")
ale_unemp  <- FeatureEffect$new(final_fit$.workflow[[1]] %>% extract_fit_parsnip(), feature="unemployment_rate", method="ale")

Conclusions & Next Steps

To integrate images, GIFs, and other media:

  1. Place your media files (PNG, JPG, GIF) in the assets/ directory of your project.
  2. Reference them with a relative path in your HTML: <img src="assets/your_image.png" alt="Description">.
  3. For animated GIFs, use the same <img> tag; browsers will render the animation automatically.
  4. Ensure you commit the assets/ folder to GitHub so that your portfolio can display them live.
